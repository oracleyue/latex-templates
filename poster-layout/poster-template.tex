	enhanced,
	left=2pt,
	right=2pt,
	colframe=#2,
	sharp corners=northwest,
	sharp corners=south,
	boxrule=6pt,
	colback=white,
	overlay={
		\def\myblock@tempa{#1}
		\ifx\myblock@tempa\@empty
		\else
		\draw[#2,fill=#2]($(frame.north west)+(.2pt,-.2pt)$)--+(.1,0)to[bend right]+(-0,-.1)--cycle;
		\node [
		anchor=south west,
		inner sep=0pt,
		outer sep=0pt
		]at(frame.north west){\titlebox[#2]{#1}};
		\fi
	},
	fontupper=\rmfamily
}
